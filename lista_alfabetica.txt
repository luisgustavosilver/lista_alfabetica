//BIBLIOTECAS
#include <stdio.h>
#include <stdlib.h>
#include <locale.h>
#include <ctype.h>
#include <string.h>

#define LINHA 20
#define COLUNA 100

//Procedimento para identificar o sistema
void clear_cls()
{
	#ifdef WIN32
		system("cls");
	#else
		system("clear");
	#endif
}
int main()
{
    setlocale(LC_ALL, " ");
    char vetstring[LINHA][COLUNA],aux[COLUNA],aux2[COLUNA],conti;
    int j,i,y, con = 0;

    for(;;) //Loop infinito
    {
        conti=' ';
        printf("digite um nome :");
        fgets(vetstring[con],COLUNA-2, stdin);
        con++;
        if(con>1)
        {
            for(j=0; j<con; j++)
            {
                strcpy(aux,vetstring[j]);
                for(y=0;y<=strlen(aux);y++)
                {
                    if(islower(aux[y]))
                    {
                        aux2[y]=toupper(aux[y]);
                    }
                    else
                    {
                         aux2[y]=aux[y];
                    }
                }
                i = j-1;
                while(i >= 0 && strcmp(vetstring[i], aux2) > 0)
                {
                    strcpy(vetstring[i+1],vetstring[i]);
                    i--;
                }
                strcpy(vetstring[i+1],aux);
            }
            printf("\nLista De nomes: \n");
            for(i = 0; i < con; i++)
            {
                puts(vetstring[i]);
            }
        }

        if(con<=19)
        {
            while(conti!='S'&& conti!='N')
            {
                printf("Deseja inserir outro nome(S/N)");
                scanf("%s",&conti);
                conti=toupper(conti);
                if (conti=='S')
                {
                    clear_cls();
                    getchar();
                }
                if(conti=='N')break;
            }
            if(conti=='N')break;
        }
    } //Fim do loop infinito

    return 0;
}
