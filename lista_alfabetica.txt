//BIBLIOTECAS
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <locale.h>
#define TAM 20

int main()
{
    //IDIOMA
    setlocale(LC_ALL, "Portuguese");
    char vetstring[TAM][100]; //declarando um vetor de 20 strings com 100 posições cada
    int i, con = 0; // declarando e inicializando a variável inteira contador

    for(;;) //loop infinito
    {
        printf("Informe o seu nome: ");
        fgets(vetstring[con], 98, stdin);

        if(con > 19)
        {
        	break;
        }

        for(i = 0; i < con; i++)
        {
        	puts(vetstring[i]);
        }
    }// fim do loop infinito

    return 0;
}
