#include <stdio.h>
#include <stdlib.h>
#include <locale.h>
#include <ctype.h>
#include <string.h>

#define LINHA 20
#define COLUNA 100


int main()
{
    //IDIOMA
    setlocale(LC_ALL, " ");
    char vetstring[LINHA][COLUNA],aux[COLUNA],conti; 
    int j,i,x, con = 0;  

    for(;;) //loop infinito
    {
        conti=' ';
        printf("digite um nome :");
        fgets(vetstring[con],COLUNA-2, stdin);

        con++;
        if(con>1)
        {
            for(j=0; j<con; j++)
            {
                strcpy(aux,vetstring[j]);
                i = j-1;
                while(i >= 0 && strcmp(vetstring[i], aux) > 0)
                {
                    strcpy(vetstring[i+1],vetstring[i]);
                    i--;
                }
                strcpy(vetstring[i+1],aux);
            }
            printf("\nLista De nomes: \n");
            for(i = 0; i < con; i++)
            {
                puts(vetstring[i]);
            }

        }
        if(con<=19)
        {
            while(conti!='S'&& conti!='N')
            {
                printf("Deseja inserir outro nome(S/N)");
                scanf("%s",&conti);
                conti=toupper(conti);
                if (conti=='S')
                {
                    system("cls");
                    getchar();
                }
                if(conti=='N')break;
            }
            if(conti=='N')break;
        }
    }// fim do loop infinito

    return 0;
}
